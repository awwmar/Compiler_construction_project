
<dataType> -> <primitiveDatatype> | TK_RUID
<primitiveDatatype> -> TK_INT | TK_REAL

<assignmentStmt> -> <singleOrRecId> TK_ASSIGNOP <expr> TK_SEM
<singleOrRecId> -> TK_ID | TK_ID TK_DOT TK_FIELDID

<expr> -> <term> <exprP>
<exprP> -> TK_PLUS <term> <exprP> | TK_MINUS <term> <exprP> | eps
<term> -> <factor> <termP>
<termP> -> TK_MUL <factor> <termP> | TK_DIV <factor> <termP> | eps
<factor> -> TK_OP <expr> TK_CL | <var>

<var> -> TK_ID | TK_NUM | TK_RNUM
